#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    int number;
    int soma = 0;
    int primo;
    int numeros[] = {1, 9, 5, 7, 9, 6, 8, 3, 9, 3};
    int vezes = 0;
    int maisnumeros[] = {10, 3, 2, 5, 2, 5, 3, 7, 8, 4};

    cout << "a)" << endl;
    cout << "numero entre [0,10]";
    cin >> number;
    while(number < 1 || number > 10){
        cout << "numero entre [0,10]";
        cin >> number;
    }
    for(int i = 0; i <= number; i++){
        soma+=i;
}
    cout << "b)" << endl;
    cout << "a soma dos inteiros entre 0 e " << number << ": " << soma << endl;

    cout << "c)" << endl;
    cout << "divisores: ";
    for (int j = 1; j < number; j++) {
        if(number%j == 0) {
            cout << j << "; ";
            primo = j;
        }
    }
    cout << endl;
    if(primo == 1){
        cout << number << " e primo" << endl;
    }
    else{
        cout << number << " nao e primo" << endl;
    }

    cout << "d)" << endl;
    for (int i = 0; i < 10; ++i) {
        if(numeros[i] == number)
            vezes++;
    }
    cout << number << " existe " << vezes << " vezes no array numeros" << endl;

    cout << "e)" << endl;
    for (int i = 0; i < 10; ++i) {
        cout << left <<  setw(5) << numeros[i] << "|" << right << setw(5) << maisnumeros[i] << "|";
        if(maisnumeros[i] > numeros[i])
            cout << " menor" << endl;
        else
            cout << " maior" << endl;
    }
    
    return 0;
}
