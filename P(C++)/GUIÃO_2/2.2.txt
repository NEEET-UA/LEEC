#include<iostream>
#include <cmath>

using namespace std;

int main(){
    double height, weight;

    cout << "Height (m) and weight (kg)? ";
    cin >> height >> weight;

    double bmi = weight / pow(height, 2.0);

    cout << "BMI: " << bmi << endl;
    
    if(bmi < 18.5) {
        cout << "skinny" << endl;
    }
    
    else if (bmi>=18.5 && bmi<25) {
        cout << "normal" << endl;
    }
    
    else if (bmi>=25 && bmi<30) {
        cout << "overweight" << endl;
    }
    
    else {
        cout << "obese" << endl;

    }

    return 0;
}
