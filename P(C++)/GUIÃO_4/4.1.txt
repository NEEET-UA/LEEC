#include <iostream>

using namespace std;

int main() {

    int n;

    cout << "Enter N" << endl;
    cin >> n;

    if (n < 1 || n > 10){
        cerr << "N INVALID" << endl;

        return 1;
    }

    int a[10];

    for (int i=0 ; i < n; i++) {
        int j = i + 1;
        
        cout << j << " ";
    }

    cout << endl;
    
    for ( int b = n -1; b >= 0; b--) {
        int r = b + 1;

        a[b] = r;
        
        cout << a[b] << " ";
    }
return 0;
}
