#include <iostream>
using namespace std;

int main() {

    int n;
    bool prime = true;

     do {
         cout << "Enter a positive integer" << endl;
         cin >> n;
         if ( n <= 0) {
             cerr << "INVALID NUMBER";
             return 1;
         }
         for(int i=2; i < n;i++) {
             if (n % i == 0) {
                prime = false;
         }
         }
         if (prime)
             cout << "PRIME NUMBER" << endl;
         else
             cout << "NOT PRIME" << endl;
     } while ( n > 0);
     
    return 0;
}
